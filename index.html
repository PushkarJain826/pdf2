arr = []
n = int(input("Enter number of elements: "))
for i in range(n):
    arr.append(int(input()))
n = len(arr)
p = 0
for i in range(n):
    for j in range(0, n - i - 1):  # Bubble Sort
        if arr[j] > arr[j + 1]:
            arr[j], arr[j + 1] = arr[j + 1], arr[j]
            p += 1
print(arr)
print("Passes:", p)

a = []
n = int(input())
for i in range(n):
    name  = input()
    marks = int(input())
    a.append((name, marks))
for i in range(n):          # selection sort (descending by marks)
    max_i = i
    for j in range(i + 1, n):
        if a[j][1] > a[max_i][1]:
            max_i = j
    a[i], a[max_i] = a[max_i], a[i]
print("Top 3:", a[:3])

#20
Find all positions using Linear Search, then sort and search using Interpolation Search. Compare results.
lst = list(map(int, input("Enter list: ").split()))
x   = int(input("Enter target: "))
pos = [i for i in range(len(lst)) if lst[i] == x]
print("Linear:", pos if pos else "Not found")
lst.sort()
l, h, idx = 0, len(lst) - 1, -1
while l <= h and lst[l] != lst[h]:
    p = l + (x - lst[l]) * (h - l) // (lst[h] - lst[l])
    if p < 0 or p >= len(lst): break
    if lst[p] == x:
        idx = p; break
    elif lst[p] < x: l = p + 1
    else: h = p - 1
print("Interpolation:", idx if idx != -1 else "Not found")

#23
Create an Employee class with bonus, retirement eligibility, and yearly salary methods. Filter and sort 5 employees.
class Employee:
    def __init__(self, n, a, d, s):
        self.name, self.age, self.dept, self.salary = n, a, d, s
    def bonus(self, p):
        self.salary += self.salary * p / 100
    def retire(self):
        return self.age >= 60
    def yearly(self):
        return self.salary * 12
emp = []
for i in range(5):
    n = input("Name: ")
    a = int(input("Age: "))
    d = input("Dept: ")
    s = float(input("Salary: "))
    e = Employee(n, a, d, s)
    e.bonus(10)
    emp.append(e)
dept = input("Enter dept to filter: ")
f = [e for e in emp if e.dept == dept]
f.sort(key=lambda x: x.salary, reverse=True)
for e in f:
    print(e.name, e.age, e.dept, e.salary, "Retire:", e.retire())
